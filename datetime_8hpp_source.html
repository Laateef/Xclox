689  static DateTime fromString(const std::string& datetime, const std::string& format)
690  {
691  int _year = 1, _month = 1, _day = 1;
692  int _hour = 0, _minute = 0, _second = 0;
693  long _subsecond = 0;
694 
695  for (size_t fmtPos = 0, dtsPos = 0; fmtPos < format.size() && dtsPos < datetime.size(); ++fmtPos) {
696  const size_t charCount = static_cast<size_t>(internal::countIdenticalCharsFrom(fmtPos, format));
697 
698  if (format[fmtPos] == '#') {
699  if (datetime[dtsPos] == '+') {
700  _year = 1;
701  ++dtsPos;
702  } else if (datetime[dtsPos] == '-') {
703  _year = -1;
704  ++dtsPos;
705  }
706  } else if (format[fmtPos] == 'y') {
707  if (charCount == 1) {
708  _year = _year * internal::readIntAndAdvancePos(datetime, dtsPos, 4);
709  } else if (charCount == 2) {
710  _year = _year * std::stoi(datetime.substr(dtsPos, charCount));
711  _year += 2000;
712  dtsPos += charCount;
713  } else if (charCount == 4) {
714  _year = _year * std::stoi(datetime.substr(dtsPos, charCount));
715  dtsPos += charCount;
716  }
717  fmtPos += charCount - 1; // skip all identical characters except the last.
718  } else if (format[fmtPos] == 'E') {
719  if (datetime.substr(dtsPos, 2) == "CE") {
720  _year = std::abs(_year);
721  dtsPos += 2;
722  } else if (datetime.substr(dtsPos, 3) == "BCE") {
723  _year = -std::abs(_year);
724  dtsPos += 3;
725  }
726  } else if (format[fmtPos] == 'M') {
727  if (charCount == 1) {
728  _month = internal::readIntAndAdvancePos(datetime, dtsPos, 4);
729  } else if (charCount == 2) {
730  _month = std::stoi(datetime.substr(dtsPos, charCount));
731  dtsPos += charCount;
732  } else if (charCount == 3) {
733  _month = internal::getShortMonthNumber(datetime.substr(dtsPos, charCount));
734  dtsPos += charCount;
735  } else if (charCount == 4) {
736  size_t newPos = dtsPos;
737  while (newPos < datetime.size() && std::isalpha(datetime[newPos]))
738  ++newPos;
739  _month = internal::getLongMonthNumber(datetime.substr(dtsPos, newPos - dtsPos));
740  dtsPos = newPos;
741  }
742  fmtPos += charCount - 1; // skip all identical characters except the last.
743  } else if (format[fmtPos] == 'd') {
744  if (charCount == 1) {
745  _day = internal::readIntAndAdvancePos(datetime, dtsPos, 2);
746  } else if (charCount == 2) {
747  _day = std::stoi(datetime.substr(dtsPos, charCount));
748  dtsPos += charCount;
749  } else if (charCount == 3) {
750  // lets the format string and the date string be in sync.
751  dtsPos += charCount;
752  } else if (charCount == 4) {
753  while (dtsPos < datetime.size() && std::isalpha(datetime[dtsPos]))
754  ++dtsPos;
755  }
756  fmtPos += charCount - 1; // skip all identical characters except the last.
757  } else if (format[fmtPos] == 'h' || format[fmtPos] == 'H') {
758  _hour = internal::readIntAndAdvancePos(datetime, dtsPos, 2);
759  fmtPos += charCount - 1; // skip all identical characters except the last.
760  } else if (format[fmtPos] == 'm') {
761  _minute = internal::readIntAndAdvancePos(datetime, dtsPos, 2);
762  fmtPos += charCount - 1; // skip all identical characters except the last.
763  } else if (format[fmtPos] == 's') {
764  _second = internal::readIntAndAdvancePos(datetime, dtsPos, 2);
765  fmtPos += charCount - 1; // skip all identical characters except the last.
766  } else if (format[fmtPos] == 'f') {
767  std::string subsecondString = datetime.substr(dtsPos, charCount);
768  _subsecond = std::stoi(subsecondString.append(9 - subsecondString.size(), '0'));
769  dtsPos += charCount;
770  fmtPos += charCount - 1; // skip all identical characters except the last.
771  } else if (format[fmtPos] == 'a' || format[fmtPos] == 'A') {
772  if (datetime.substr(dtsPos, 2) == "pm" || datetime.substr(dtsPos, 2) == "PM") {
773  _hour = (_hour > 12 ? _hour : _hour + 12);
774  dtsPos += 2;
775  }
776  } else {
777  // not a pattern, skip it in the date string.
778  ++dtsPos;
779  }
780  }
781 
782  return DateTime(Date(_year, _month, _day), Time(_hour, _minute, _second, Nanoseconds(_subsecond)));
783  }
