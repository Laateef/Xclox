153  inline bool isPattern(char flag, size_t count = 0)
154  {
155  static const std::unordered_map<char, size_t> patternMap {
156  { '#', 1 },
157  { 'E', 1 },
158  { 'y', (1 | 1 << 1 | 1 << 3) },
159  { 'M', (1 | 1 << 1 | 1 << 2 | 1 << 3) },
160  { 'd', (1 | 1 << 1 | 1 << 2 | 1 << 3) },
161  { 'h', (1 | 1 << 1) },
162  { 'H', (1 | 1 << 1) },
163  { 'm', (1 | 1 << 1) },
164  { 's', (1 | 1 << 1) },
165  { 'f', (1 | 1 << 1 | 1 << 2 | 1 << 3 | 1 << 4 | 1 << 5 | 1 << 6 | 1 << 7 | 1 << 8) },
166  { 'a', 1 },
167  { 'A', 1 }
168  };
169  const auto iter = patternMap.find(flag);
170  return iter != patternMap.cend() && (count == 0 || iter->second & (1 << count - 1));
171  }
