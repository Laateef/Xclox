18  auto compare = [](const DateTime& dt, int year, int month, int day, int hour, int minute, int second, int milli, int micro, int nano) {
19  return dt.year() == year && dt.month() == month && dt.day() == day
20  && dt.hour() == hour && dt.minute() == minute && dt.second() == second
21  && dt.millisecond() == milli && dt.microsecond() == micro && dt.nanosecond() == nano;
22  };

204  TEST_CASE("addition & subtraction")
205  {
206  SUBCASE("add duration")
207  {
208  // Adding time durations which don't affect the date part.
209  CHECK(DateTime(Date(2045, 3, 27), Time(16, 2, 3, 4)).addDuration(DateTime::Minutes(10)) == DateTime(Date(2045, 3, 27), Time(16, 12, 3, 4)));
210  CHECK(DateTime(Date(2045, 3, 27), Time(1, 2, 3, 4)).addDuration(DateTime::Hours(3)) == DateTime(Date(2045, 3, 27), Time(4, 2, 3, 4)));
211  // Adding time durations which affect the date part.
212  CHECK(DateTime(Date(2045, 3, 27), Time(16, 2, 3, 4)).addDuration(DateTime::Hours(10)) == DateTime(Date(2045, 3, 28), Time(2, 2, 3, 4)));
213  // Adding date durations which only affect the date part.
214  CHECK(DateTime(Date(2045, 3, 27), Time(1, 2, 3, 4)).addDuration(DateTime::Days(3)) == DateTime(Date(2045, 3, 30), Time(1, 2, 3, 4)));
215  // Adding a negative duration
216  CHECK(DateTime(Date(2045, 3, 27), Time(1, 2, 3, 4)).addDuration(-DateTime::Days(3)) == DateTime(Date(2045, 3, 24), Time(1, 2, 3, 4)));
217  }
218  SUBCASE("subtract duration")
219  {
220  // Subtracting time durations which don't affect the date part.
221  CHECK(DateTime(Date(2045, 3, 27), Time(16, 2, 3, 4)).subtractDuration(DateTime::Minutes(10)) == DateTime(Date(2045, 3, 27), Time(15, 52, 3, 4)));
222  CHECK(DateTime(Date(2045, 3, 27), Time(7, 2, 3, 4)).subtractDuration(DateTime::Hours(3)) == DateTime(Date(2045, 3, 27), Time(4, 2, 3, 4)));
223  // Subtracting time durations which affect the date part.
224  CHECK(DateTime(Date(2045, 3, 27), Time(6, 2, 3, 4)).subtractDuration(DateTime::Hours(10)) == DateTime(Date(2045, 3, 26), Time(20, 2, 3, 4)));
225  // Subtracting date durations which only affect the date part.
226  CHECK(DateTime(Date(2045, 3, 27), Time(1, 2, 3, 4)).subtractDuration(DateTime::Days(3)) == DateTime(Date(2045, 3, 24), Time(1, 2, 3, 4)));
227  // Subtracting a negative duration
228  CHECK(DateTime(Date(2045, 3, 27), Time(1, 2, 3, 4)).subtractDuration(-DateTime::Days(3)) == DateTime(Date(2045, 3, 30), Time(1, 2, 3, 4)));
229  }
